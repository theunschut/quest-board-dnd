```csharp
public interface IImageValidationService
{
    // Validates an original (required-if-present) + optional cropped image pair against the shared
    // MIME allowlist, extension allowlist, and size limit. Returns errors keyed by field name (empty = valid).
    IList<ImageValidationError> ValidateImagePair(
        ImageFileInput? original, ImageFileInput? cropped);
}

public record ImageFileInput(long Length, string ContentType, string FileName, string FieldName);
public record ImageValidationError(string FieldName, string Message);
```
